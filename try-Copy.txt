<script>
document.getElementById("btn").addEventListener("click",getdata);
const ori = document.querySelector('#pick')
const dist = document.querySelector('#drop')
const getdata = () => {
  const url = `http://localhost:7000/${ori.value}/${dist.value}`
  fetch(url)
    .then((response) => !response.ok ? null : response.json()
          .then((data) => {
    const t = data.time
	const d = data.dist
	document.getElementById("distance").innerHTML=d
    document.getElementById("time").innerHTML=t;    
  })
         )
}
</script>
